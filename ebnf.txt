# #  comment
# () grouping
# [] optional
# *  zero or more
# +  one or more
# |  alternative

program ::= (block)*
block ::= (vardecl)* (stmt)*

vardecl ::= "int" IDENT ";" |
             "int" IDENT "=" INT_LITERAL;

stmt ::= block
        | "while" "(" exp ")" stmt
        | "if" "(" exp ")" stmt ["else" stmt]
        | IDENT "=" exp

exp ::= "(" exp ")"
        | (IDENT | INT_LITERAL)
        | exp (">" | "<" | "!=" | "==" | "+" | "-" | "/" | "*" | "||" | "&&") exp # binary operators
